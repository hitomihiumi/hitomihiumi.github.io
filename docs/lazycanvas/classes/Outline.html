Class Outline

Example

const { LazyCanvas, Outline, CircleLayer } = require('@hitomihiumi/lazy-canvas')

let outline = new Outline()
.setType('center')
.setColor('#ff8a8a')
.setStroke(2)

let circle = new CircleLayer()
.setX(100)
.setY(200)
.setRadius(50)
.setColor('#fff')
.setOutline(
    outline
)

const lazy = new LazyCanvas()
.addLayers(circle)
