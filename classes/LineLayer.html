Class LineLayer

Example

const { LazyCanvas, LineLayer } = require('@hitomihiumi/lazy-canvas')

let line = new LineLayer()
.setPoints({ x: 10, y: 10 }, { x: 100, y: 100 })
.setColor('#fff')
.setStroke(5)
.setLineDash([10, 5])

const lazy = new LazyCanvas()
.addLayers(line)
//...
